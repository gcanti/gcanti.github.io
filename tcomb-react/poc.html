var React = require('react');
var t = require('tcomb-react');
var Playground = require('tcomb-react/Playground.jsx');

var AlertType = t.enums.of('info success danger', 'AlertType');

var BootstrapAlert = React.createClass({

  // a tcomb struct
  // instead of propTypes
  mixins: [t.react.Mixin({
    type: AlertType,
    content: t.react.ReactElement
  })],

  render: function () {
    return <div className={'alert alert-' + this.props.type}>{this.props.content}</div>;
  }

});

var playground = React.render(
  <Playground
    component={BootstrapAlert}
    props={{type: 'danger', content: <b>You can even use JSX in the form</b>}} />,
  document.getElementById('app')
);

playground.show();
